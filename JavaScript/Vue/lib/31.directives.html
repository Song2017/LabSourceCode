自定义指令 Vue.directive('focus', {})

// 全局指令 v-focus, 使用时默认添加v-
// 参数1, 指令名称, 此时不需要添加v-
// 参数2, 对象, 包括了指令的钩子函数, 会在特定的阶段调用
Vue.directive('focus', {
// el是原生的DOM对象
bind: function(el){}, // 指令绑定到元素的时候 触发1次
inserted:function(el){}, // 元素插入到DOM的时候 触发1次
updated:function(el){} //VNode更新的时候 触发n次
});
< input type="text" name="itext" id="input1" v-focus>
